4. Memo
  Router
  / -> Home
  /join -> Join
  /login -> Login
  /search -> Search

  /users/:id -> See User
  /users/logout -> Log out
  /users/edit -> Edit User
  /users/remove -> Remove User

  /videos/:id -> See Video
  /videos/:id/edit -> Edit Video
  /videos/:id/remove -> Remove Video
  /videos/upload -> Upload Video
